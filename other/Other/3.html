当然, 最终还是归于基础

var arr = [ [1, 2, 2], [3, 4, 5, 5], [6, 7, 8, 9, [11, 12, [12, 13, [14] ] ] ], 10];

// 扁平化
function flat(arr){
    while (arr.some(Array.isArray)) {
        arr = [].concat(...arr);
    }
    return arr;
}

// 去重
function unique(arr) {
    return arr.filter((item, index) => arr.indexOf(item) === index);
}

// 排序
function sort(arr){
    return arr.sort((a, b) => a - b)
}
``
