# URL参数解析

// https://raindays.cn?name=libai&age=20&job=teacher
function urlParse(url) {
    const res = new Object();
    const arr = url.substr(url.indexOf('?') + 1).split('&');
    arr.forEach(item => {
        let tmp = item.split('=');
        res[tmp[0]] = tmp[1]
    })
    return res
}
